
For multi polygon state :

const multiPolygon = db.states.findOne({ STUSPS: 'CO' }).geometry.coordinates;
multiPolygon.forEach(polygon => {
    const sightingsWithinPolygon = db.sightings2.find({
        location: {
            $geoWithin: {
                $geometry: {
                    type: "Polygon",
                    coordinates: polygon
                }
            }
        }
    });

    sightingsWithinPolygon.forEach(sighting => {
        printjson(sighting);
    });
});
